# ============================================================
# machine_runtime_monitor_top.xdc  (Blynk/ESP32-enabled version)
# Target : Xilinx XC7A35T-FTG256-1  (AT-STLN-ARTIX 7-001)
# Voltage : 3.3 V LVCMOS33 on all user banks
# ============================================================

# ------------------------------------------------------------
# Clock - 24 MHz ASDM1 oscillator (Bank 15, MRCC pin D13)
# ------------------------------------------------------------
set_property PACKAGE_PIN D13      [get_ports clk_24mhz]
set_property IOSTANDARD LVCMOS33  [get_ports clk_24mhz]
create_clock -period 41.667 -name sys_clk [get_ports clk_24mhz]

# Switch/button/UART inputs are asynchronous - no timing needed
set_false_path -from [get_ports {rst sw0_start sw1_stop sw2_fault \
                                  sw3_mode0 sw4_mode1 sw5_mode2 \
                                  esp_uart_rx}]

# ------------------------------------------------------------
# Reset - 4x4 Matrix Keypad, Button "0"  Pin A13  (Bank 35, active HIGH)
# ------------------------------------------------------------
set_property PACKAGE_PIN A13      [get_ports rst]
set_property IOSTANDARD LVCMOS33  [get_ports rst]

# ------------------------------------------------------------
# Slide Switches (Bank 35)
# ------------------------------------------------------------
set_property PACKAGE_PIN C9       [get_ports sw0_start]
set_property IOSTANDARD LVCMOS33  [get_ports sw0_start]

set_property PACKAGE_PIN B9       [get_ports sw1_stop]
set_property IOSTANDARD LVCMOS33  [get_ports sw1_stop]

set_property PACKAGE_PIN G5       [get_ports sw2_fault]
set_property IOSTANDARD LVCMOS33  [get_ports sw2_fault]

set_property PACKAGE_PIN A7       [get_ports sw3_mode0]
set_property IOSTANDARD LVCMOS33  [get_ports sw3_mode0]

set_property PACKAGE_PIN C7       [get_ports sw4_mode1]
set_property IOSTANDARD LVCMOS33  [get_ports sw4_mode1]

set_property PACKAGE_PIN A10      [get_ports sw5_mode2]
set_property IOSTANDARD LVCMOS33  [get_ports sw5_mode2]

# ------------------------------------------------------------
# Relay - Bank 34, pin L5  (drives SS8050 NPN -> 5 V relay coil)
# HIGH = relay energised
# ------------------------------------------------------------
set_property PACKAGE_PIN L5       [get_ports relay]
set_property IOSTANDARD LVCMOS33  [get_ports relay]

# ------------------------------------------------------------
# Buzzer - Bank 35, pin K5  (drives BC817 NPN -> active piezo)
# HIGH = buzzer ON
# ------------------------------------------------------------
set_property PACKAGE_PIN K5       [get_ports buzzer]
set_property IOSTANDARD LVCMOS33  [get_ports buzzer]

# ------------------------------------------------------------
# MAX7219 SPI  (Bank 15, level-shifted to 5 V via U17/U18/U19)
# ------------------------------------------------------------
set_property PACKAGE_PIN J15      [get_ports seg_din]
set_property IOSTANDARD LVCMOS33  [get_ports seg_din]

set_property PACKAGE_PIN J16      [get_ports seg_cs]
set_property IOSTANDARD LVCMOS33  [get_ports seg_cs]

set_property PACKAGE_PIN H12      [get_ports seg_clk]
set_property IOSTANDARD LVCMOS33  [get_ports seg_clk]

# ------------------------------------------------------------
# NEW: ESP32 UART link via PMOD connector J16 (Bank 34)
#   IO_0 (T2) = FPGA UART TX  -> wire to ESP32 GPIO20 (RX), J13 pin 5
#   IO_1 (R3) = FPGA UART RX  <- wire to ESP32 GPIO21 (TX), J13 pin 6
#   Both sides are 3.3V - direct jumper wire, no level shifting needed.
# ------------------------------------------------------------
set_property PACKAGE_PIN T2       [get_ports esp_uart_tx]
set_property IOSTANDARD LVCMOS33  [get_ports esp_uart_tx]

set_property PACKAGE_PIN R3       [get_ports esp_uart_rx]
set_property IOSTANDARD LVCMOS33  [get_ports esp_uart_rx]

# ------------------------------------------------------------
# Bitstream config - mandatory for Artix-7 DRC clean pass
# ------------------------------------------------------------
set_property BITSTREAM.GENERAL.COMPRESS         TRUE [current_design]
set_property BITSTREAM.CONFIG.SPI_BUSWIDTH      4    [current_design]
set_property CONFIG_VOLTAGE                     3.3  [current_design]
set_property CFGBVS                             VCCO [current_design]
