//LED Blinking with Clock Output
// This module provides a 1-second LED blinking pattern and passes the input clock through to an external output pin.





module led_blinking(clk,clk_out,led);
input clk;
output reg led=0;
output  clk_out;

assign clk_out=clk;

reg [26:0]counter=0;

always @(posedge clk)begin
   if(counter==27'd23_999_999)begin
      counter = 27'd0;
	  led=~led;
   end else
      counter=counter +1;
end
endmodule
Tcl
# Clock
set_property PACKAGE_PIN D13 [get_ports clk]
set_property IOSTANDARD LVCMOS33 [get_ports clk]
create_clock -period 41.667 [get_ports clk]

# Output Assignments
set_property PACKAGE_PIN T2 [get_ports clk_out]
set_property PACKAGE_PIN T3 [get_ports led]
set_property IOSTANDARD LVCMOS33 [get_ports clk_out]
set_property IOSTANDARD LVCMOS33 [get_ports led]
